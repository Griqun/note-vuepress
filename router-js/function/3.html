    Invite，List是两个内容不同页面的组件，immediate-check禁止初始化执行




     

















































































    <div class="content">
    <!-- 内容 -->
    <van-pull-refresh v-model="refreshing" @refresh="onRefresh" class="refresh-class">
      <van-list
        v-model:loading="loading"
        :offset="10"
        :finished="finished"
        finished-text="没有更多了"
        @load="onLoad"
        :immediate-check="false"
      >
        <Invite v-if="active == 0" :list="list" />
        <List v-if="active == 1" :list="list" @refreshHandel="refreshHandel" />
      </van-list>
    </van-pull-refresh>
</div>
<script lang="ts" setup>
import List from './list.vue';
import Invite from './invite.vue';
const loading = ref(false); //是否处于加载状态
const finished = ref(false); //是否已加载完成
const refreshing = ref(false); //是否处于下拉刷新状态
const list = ref<inviteList[]>([]);
const listQuery = reactive({
	pageNum: 1, // 当前页码数
	pageSize: 10, // 一页显示10条
});
const active = ref(0);
// 获取列表
const getMessage = async (type?: boolean) => {
	let params: inviquery = {
		shopowner: store.userInfo.phone,
		...listQuery,
	};
	try {
		const { code, data } = await post(
			active.value == 0 ? api.shop.getStoreMessage : active.value == 1 ? api.shop.getByShopownerShop : '',
			params,
			false
		);
		console.log('res', params, data);
		if (code == 200) {
      {/* 页数加1的时候 type为true，需要数据拼接，否则直接赋值显示 */}
			list.value = type ? [...list.value, ...data.records] : (list.value = data.records);
			console.log('list2', list.value);
      {/* 总数小于等于列表的总长度时，表明加载完成 */}
			if (data.total <= list.value.length) {
				finished.value = true;
			}
			loading.value = false;
		}
	} catch (error) {
		loading.value = false;
		finished.value = true;
		console.log(error);
	}
};
{/* 立即执行 */}
getMessage();
// 上拉加载
const onLoad = () => {
	console.log('触发');
	loading.value = true;
	listQuery.pageNum += 1;
	getMessage(true);
};
{/* // 下拉刷新 ，下拉仅仅显示加载中，其它状态都关闭 */}
const onRefresh = () => {
	console.log('下拉');
	finished.value = false;
	loading.value = true;
	listQuery.pageNum = 1;
	getMessage();
	refreshing.value = false;
};
{/* // 店员名单刷新列表 该子组件发射监听的方法，emits('refreshHandel') */}
const refreshHandel = () => {
	console.log('刷新');
	finished.value = false;
	loading.value = true;
	listQuery.pageNum = 1;
	getMessage();
};
</script>
