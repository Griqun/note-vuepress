    <template>
     <!-- 加入购物车 -->
    <van-action-bar>
      <van-action-bar-icon icon="chat-o" text="客服" dot />
      <van-action-bar-icon icon="cart-o" text="购物车" :badge="countLists" @click="toCartList" />
      <van-action-bar-icon icon="shop-o" text="店铺" badge="12" />
      <van-action-bar-button type="warning" text="加入购物车" @click="show = true" />
      <van-action-bar-button type="danger" text="立即购买" />
    </van-action-bar>
    <!-- 筛选弹出层 -->
    <van-action-sheet v-model:show="show" :round="false">
      <div class="sheet-class">
        <div class="head">
          <div class="title-class">
            <van-image width="100" height="100" src="https://fastly.jsdelivr.net/npm/@vant/assets/cat.jpeg" />
            <div class="size">
              <span>￥12.00</span>
              <span>库存10000件</span>
              <span>已选规格：{{ state.selected.title }}</span>
            </div>
          </div>
          <div @click="show = false"><van-icon name="cross" /></div>
        </div>
        <div class="category">
          <h5>选择类别</h5>
          <span @click="choose(item, index)" :class="[item.check ? 'active' : '']" v-for="(item, index) in state.cate"
            ><van-image width="30" height="30" class="img" :src="item.image" />{{ item.title }}</span
          >
        </div>
        <div class="foot">
          <div class="number">购买数量<span style="color: red">（限购12件）</span></div>
          <div class="count">
            <span @click="add(0)" :style="{ background: count === 1 ? '#f6f5f5' : '#eee' }"><van-icon name="minus" /></span>
            <span>{{ count }}</span>
            <span @click="add(1)"><van-icon name="plus" /></span>
          </div>
        </div>
        <van-button color="orange" round class="btn" @click="addCard">确定</van-button>
      </div>
    </van-action-sheet>
</template>

    <script lang="ts" setup>
import { computed, reactive, ref } from "@vue/reactivity";
import { Toast } from "vant";
import { useRouter } from "vue-router";
import { useStore } from "../../store/modules/user";
const store = useStore();
const router = useRouter();
const show = ref(false);
const count = ref(1);
const numbers = ref<any>("");
const countLists = computed(() => store.count);
interface Cate {
  title: string;
  image: string;
  shop: string;
  id: number;
  price: string;
  check?: boolean;
  count?: number;
}
interface Query {
  cate: Cate[];
  selected: Cate;
}
const state = reactive(<Query>{
  cate: [
    { title: "爱奇艺", image: "", shop: "店铺1", id: 1, price: "10.98" },
    { title: "优酷视频", image: "", shop: "店铺1", id: 2, price: "20.58" },
  ],
  selected: {},
});
const active = ref(-1);
// 给所有的数据先添加一个是否选择的字段check，真实请求接口时在接口处赋值之后添加
for (let i = 0; i < state.cate.length; i++) {
  state.cate[i].check = false;
}
// 或者用下面写法
// state.cate.map((item, index) => {
//   item.check = false;
// });
// 选择切换规格
const choose = (item: Cate, index: number) => {
  active.value = index;
  count.value = 1;
  state.cate.map((i: Cate) => {
    // 直接把整条数据判断，接口需要传整条数据的情况下
    if (i == item) state.cate[index].check = !state.cate[index].check;
    else i.check = false;
  });
  // 再把选择的这条数据添加到最终的列表
  if (item.check) {
    state.selected = item;
  } else {
    state.selected = {} as Cate;
  }
};
// 数量相加减
const add = (index: number) => {
  switch (index) {
    case 0:
      if (count.value <= 1) return;
      count.value--;
      break;
    case 1:
      count.value++;
      break;
    default:
      break;
  }
};
// 确定规格
const addCard = () => {
  if (Object.getOwnPropertyNames(state.selected).length == 0) {
    Toast("请选择规格");
    return;
  }
  show.value = false;
  if (numbers.value == "") numbers.value = 1;
  else numbers.value++;
  Toast("加入购物车成功");
  console.log("state.selected", state.selected, count.value);
  // 把购物车信息存在store
  state.selected.count = count.value; // 把自己的数量也放在列表里
  store.goodsList(state.selected);
  console.log("可获取的数据", store.cartList);

  addAxios(state.selected.id);
};
const addAxios = async (id: number) => {
  try {
    // 请求
    setTimeout(() => {
      console.log("添加购物车请求成功", id);
    }, 1500);
  } catch (error) {}
};
const toCartList = () => {
  let data = JSON.stringify({
    list: state.selected,
    count: count.value,
    numbers: numbers.value,
  });
  router.push({ name: "list", params: { data } });
};
</script>
