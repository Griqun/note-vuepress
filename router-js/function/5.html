    // currentIndex -1：都不亮  0 ：上箭头亮 1 ：下箭头亮
    screenList: {
      nameKey: "saleNum",
      currentIndex: -1,
      data: [
        {
          name: "销量",
          status: 1,
          key: "saleNum",
        },
        {
          name: "价格",
          status: -1,
          key: "priceFee",
        },
      ],
    },

    screenTap({ target }) {
    wx.showLoading({
      title: "加载中...",
      mask: true,
    });
    let index = target.dataset.index;  //获取当前点击了哪一条（销量or价格）
    this.data.screenList.currentIndex = index;
    let item = this.data.screenList.data[index];  //另存点击选中的这条
    this.setData({
      ["screenList.nameKey"]: item.key,  //赋值点击的keyname
    });
    this.data.screenList.data.forEach((v, i) => {
      if (i != index) {    //当数据的index不等于点击选中的index，就让其还原到初始值-1
        v.status = -1; //取消所有的值
      }
    });
    if (index == this.data.screenList.currentIndex) {   //当点击选中的等于上一次选中的（当点了价格 == 1，再点一次也是1）
      item.status = item.status == 0 ? 1 : 0;  //如果是0上箭头亮，这时候再点一次就应该是1下箭头亮，否则相反为0，互相切换
      if (item.key == "saleNum") {
        switch (item.status) {
          case 0:
            this.setData({
              productList: this.data.productList.sort((a, b) => a.saleNum - b.saleNum),
            });
            break;
          case 1:
            this.setData({
              productList: this.data.productList.sort((a, b) => b.saleNum - a.saleNum),
            });
            break;
          default:
            break;
        }
      } else {
        switch (item.status) {
          case 0:
            this.setData({
              productList: this.data.productList.sort((a, b) => b.priceFee - a.priceFee),
            });
            break;
          case 1:
            this.setData({
              productList: this.data.productList.sort((a, b) => a.priceFee - b.priceFee),
            });
            break;
          default:
            break;
        }
      }
      wx.hideLoading();
    }
    this.setData({
      ["screenList.data"]: this.data.screenList.data,  //最后一步要改变数据
    });
  },
